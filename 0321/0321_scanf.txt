#pragma warning(disable : 4996)
#include <stdio.h>

int main(void) 
{
	// scanf
	// scanf(서식문자열, 주소)
	// 입력이 들어올때 까지 대기하다가 들어오면 해당 주소에 값을 써넣음
	int n;
	printf("Enter integer value");
	scanf_s("%d", &n);
	printf("n : %d", n);
	// & : 주소 연산자
	// 변수 앞에 붙어서 해당 변수의 주소를 반환하는 연산자

	float f;
	scanf_s("%f", &f);
	
	char str[20];
	scanf("%s", str);
	//scanf_s("%s", str, 10 * sizeof(char));
	// sizeof 키워드
	// 사용형태 : sizeof(자료형)
	// 자료형에 해당하는 데이터 크기를 반환함

	printf("str : %s", str);

	return 0;
}
